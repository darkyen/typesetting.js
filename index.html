Usage not to be confused with Unicorns

Usage is pretty simple, Just include the typesetting.js and Sizzle†1 and you can start controlling your text like a boss!

typesetting.js splits up your text into easy to control span elements with simple incremental classes. 

<h1 class="title-text">Hello World!</h1>
<script>
typesetting.letters('.title-text');
</script>

will produce

<h1 class="title-text">
	<span class="char1">H</span>
	<span class="char2">e</span>
	<span class="char3">l</span>
	<span class="char4">l</span>
	<span class="char5">o</span>
	<span class="char6"></span>
	<span class="char7">W</span>
	<span class="char8">o</span>
	<span class="char9">r</span>
	<span class="char10">l</span>
	<span class="char11">d</span>
	<span class="char12">!</span>
</h1>

Functions for you to work with:

typesetting.letters(selector); will split the text up by letter, treating whitespace as a letter. Classname: char1
typesetting.words(selector); will split the text up by word.†2 Classname: word2
typesetting.lines(selector); will split the text up by line on the <br /> tag. Classname: line3

use The Source Luke

var typesetting = (function() {
    function injector(t, splitter, klass, after) {
        var a = (t.innerText || t.textContent),
            inject = '';
        a = a.split(splitter);
        if (a.length) {
            a.forEach(function(item, i) {
                inject += '' + item + '' + after;
            });
            t.innerHTML = inject; 
        }
    }
    function replaceNodeWith(element, search, replace) {
        var elements = element.childNodes;
        for (var i = 0, l = elements.length; i < l; i++) {
            if (elements[i].tagName === search) {
                element.replaceChild(document.createTextNode(replace), elements[i]);
            }
        }
    }
    var typesetting = {
        letters: function(queryString, context) {
            context = context || document;
            var elements = Sizzle(queryString, context);
            elements.forEach(function(item) {
                injector(item, '', 'char', '');
            });
            return elements;
        },
        lines: function(queryString, context) {
            context = context || document;
            var elements = Sizzle(queryString, context);
            elements.forEach(function(item) {
                var r = "eefec303079ad17405c889e092e105b0";
                replaceNodeWith(item, 'BR', r);
                injector(item, r, 'line', '');
            });
            return elements;
        },
        words: function(queryString, context) {
            context = context || document;
            var elements = Sizzle(queryString, context);
            elements.forEach(function(item) {
                injector(item, ' ', 'word', '');
            });
            return elements;
        }
    };  
    return typesetting;
})();

Footnotes

1: Sizzle dependancy will be removed in the next version.

2: Whitespace by default is not retained between the words however this can easily be modified by changing the after parameter in the words() function.
